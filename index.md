function closeTabs() {
	var i;
	var x = document.getElementsByClassName("tab");
	for (i = 0; i < x.length; i++) {
		x[i].style.display = "none";
	}
}

function openTab(tabName) {
	closeTabs();
	document.getElementById(tabName).style.display = "block";
}
